# ══════════════════════════════════════════════════════════════
#  Display helpers

CATEGORY_COLORS = {
    "INDIA PII":            "\033[91m",   # red
    "CLOUD SECRETS":        "\033[93m",   # yellow
    "DATABASE":             "\033[95m",   # magenta
    "CODE + SECRETS":       "\033[96m",   # cyan
    "FINANCIAL":            "\033[91m",   # red
    "CLEAN — NO VIOLATION": "\033[92m",   # green
}
RESET  = "\033[0m"
BOLD   = "\033[1m"
DIM    = "\033[2m"


def color(text: str, cat: str) -> str:
    return CATEGORY_COLORS.get(cat, "") + text + RESET


def print_menu():
    print(f"\n{BOLD}{'='*65}{RESET}")
    print(f"{BOLD}  BLIP SENTINEL — Live Judge Demo{RESET}")
    print(f"{DIM}  Daemon must be running: python main.py{RESET}")
    print(f"{'='*65}{RESET}\n")

    current_cat = ""
    for t in TESTS:
        if t["category"] != current_cat:
            current_cat = t["category"]
            print(f"  {color(f'── {current_cat} ──', current_cat)}")

        score_str  = (f"score ~{t['score']}"
                      if t["score"] != "0" else "CLEAN")
        sev_str    = t["severity"]
        print(
            f"  [{BOLD}{t['id']}{RESET}] "
            f"{t['label']:<42} "
            f"{DIM}{sev_str:<10}{RESET} "
            f"{DIM}{score_str}{RESET}"
        )

    print(f"\n  [{BOLD}A{RESET}]  AUTO MODE — cycle all with 10s gap")
    print(f"  [{BOLD}AC{RESET}] AUTO CRITICAL — cycle violations only (skip clean)")
    print(f"  [{BOLD}Q{RESET}]  Quit")
    print(f"\n{'='*65}\n")


def copy_and_show(t: dict, pause: bool = True):
    """Copy payload and print what to expect."""
    print(f"\n  {BOLD}[{t['id']}] {t['label']}{RESET}")
    print(f"  {DIM}Category : {t['category']}{RESET}")
    print(f"  {DIM}Severity : {t['severity']}  |  Score: ~{t['score']}{RESET}")
    print(f"  {DIM}Payload  : {t['text'][:72].strip()}...{RESET}"
          if len(t['text']) > 72
          else f"  {DIM}Payload  : {t['text'].strip()}{RESET}")
    print(f"\n  {BOLD}► Copying to clipboard...{RESET}")
    pyperclip.copy(t["text"])
    print(f"  {BOLD}✓ Copied! ({len(t['text'])} chars){RESET}")
    print(f"\n  {BOLD}Expected:{RESET} {t['expect']}\n")
    print(f"  {DIM}{'─'*55}{RESET}")

    if pause:
        input(f"  {DIM}Press Enter for menu...{RESET}")


def auto_mode(violations_only: bool = False):
    subset = (
        [t for t in TESTS if t["score"] != "0"]
        if violations_only else TESTS
    )
    label = "CRITICAL ONLY" if violations_only else "ALL"
    print(f"\n  {BOLD}AUTO MODE [{label}]{RESET} — 10s between payloads")
    print(f"  {DIM}Press Ctrl+C to stop{RESET}\n")

    cycle = 0
    while True:
        cycle += 1
        print(f"\n  {DIM}── Cycle {cycle} ──{RESET}")
        for t in subset:
            copy_and_show(t, pause=False)
            for i in range(10, 0, -1):
                print(f"\r  {DIM}Next in {i}s...  {RESET}", end="", flush=True)
                time.sleep(1)
            print()


def interactive_mode():
    print_menu()
    while True:
        try:
            choice = input(f"  {BOLD}Select [{'/'.join([t['id'] for t in TESTS[:5]])}... A/AC/Q]: {RESET}").strip().upper()
        except (KeyboardInterrupt, EOFError):
            print(f"\n  {DIM}Goodbye!{RESET}\n")
            break

        if choice == "Q":
            print(f"\n  {DIM}Goodbye!{RESET}\n")
            break
        elif choice == "A":
            try:
                auto_mode(violations_only=False)
            except KeyboardInterrupt:
                print(f"\n  {DIM}Auto mode stopped{RESET}")
                print_menu()
        elif choice == "AC":
            try:
                auto_mode(violations_only=True)
            except KeyboardInterrupt:
                print(f"\n  {DIM}Auto mode stopped{RESET}")
                print_menu()
        else:
            match = next((t for t in TESTS if t["id"] == choice.zfill(2)), None)
            if match:
                copy_and_show(match, pause=True)
                print_menu()
            else:
                print(f"  {DIM}Invalid choice — enter 01–20, A, AC, or Q{RESET}")


# ══════════════════════════════════════════════════════════════

if __name__ == "__main__":
    parser = argparse.ArgumentParser(
        description="Blip Sentinel — Live Judge Demo"
    )
    parser.add_argument(
        "--auto", action="store_true",
        help="Auto-cycle all payloads with 10s gap"
    )
    parser.add_argument(
        "--critical", action="store_true",
        help="Auto-cycle violation payloads only"
    )
    parser.add_argument(
        "--id", type=str,
        help="Copy a specific test directly e.g. --id 07"
    )
    args = parser.parse_args()

    if args.id:
        match = next(
            (t for t in TESTS if t["id"] == args.id.zfill(2)), None
        )
        if match:
            copy_and_show(match, pause=False)
        else:
            print(f"  Test ID {args.id} not found")
    elif args.auto:
        auto_mode(violations_only=False)
    elif args.critical:
        auto_mode(violations_only=True)
    else:
        interactive_mode()
